\subsection{Mother institution logo}
\begin{frame}[fragile]{Theme options}{Mother institution logo} %option fragile needed for verbatim environment
To select the logo of the mother institution the options are:
\begin{itemize}
	\item \texttt{kul} \qquad for KU Leuven \includegraphics[height=10px]{graphics/kul}
	\item \texttt{vito} \quad for VITO \includegraphics[height=10px]{graphics/vito}
	\item \texttt{imec} \qquad for  IMEC \includegraphics[height=10px]{graphics/imec}
	\item \texttt{uhasselt} \qquad for UHasselt \includegraphics[height=10px]{graphics/uhasselt}

\begin{verbatim}
    \usetheme[kul]{energyville}
\end{verbatim}
\end{frame}

\section{Title page}
\begin{frame}[fragile]{Title page logos}
By default, the title page defines \texttt{titlegraphic} as EnergyVille logo. A custom graphic can be defined with:

\begin{verbatim}
    \titlegraphic{ \includegraphics[height=45px]{mytitlelogo.png} }
\end{verbatim}

If you would like to keep both logos, the following snippet can be used.

\begin{verbatim}
    \titlegraphic{
        \hfill
        \includegraphics[height=45px]{graphics/energyville}
        \hspace{40px} % you might need to adjust this space
        \includegraphics[height=45px]{mytitlelogo.png}
        \hfill
    }
\end{verbatim}
\end{frame}


\section{Additional commands}
\begin{frame}[fragile]{Additional commands}
Three additional commands are provided in the template.

Text indicating the level of confidentiality (will appear on the bottom left)
\begin{verbatim}
    \confidentiality{Confidential}
\end{verbatim}

Logo of the project (will appear on the top right)
\begin{verbatim}
    \projectlogo{\includegraphics[height=15px]{graphics/project}}
\end{verbatim}

Logo of the funding body (will appear on the top right under the project logo)
\begin{verbatim}
    \fundinglogo{\includegraphics[height=15px]{graphics/funding}}
\end{verbatim}

\end{frame}

\section{Environments}
\begin{frame}[fragile]{Environments}{Equations and math}   %option fragile needed for verbatim environment
Equations and other mathematical symbols use serif typeface:
\begin{gather*}
f(x)= a x^2 + b x + c
\end{gather*}

Style of individual symbols can be changed manually:
\begin{gather*}
\mathbf{\hat{\beta}} = \underset{b}{\rm arg\,min}\,(\mathbf{y}- \mathbf{X} \mathbf{b})^{\mathsf{T}}\,\boldsymbol{\Omega}^{-1}(\mathbf{y}- \mathbf{X} \mathbf{b}) \\
\mathsf{    G_t = \alpha e^{-\beta e^{-\gamma \cdot t}}    }
\end{gather*}

To change all math into sans-serif delete this line from the preamble in .tex file:
	\begin{verbatim}
	\usefonttheme[onlymath]{serif}
	\end{verbatim}
\end{frame}
